def _parse_thchsdict(file):
    thchsdict = {}
    for line in file:
        parts = line.strip('\n').split(' ')
        pinyin = parts[0]
        phoneme = ' '.join(parts[1:])
        thchsdict[pinyin] = phoneme
    return thchsdict
